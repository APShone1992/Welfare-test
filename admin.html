Supabase setup (free, 3 minutes)

  1. Go to supabase.com → sign in with GitHub → New project
  2. Wait ~2 min, then open SQL Editor and run: create table sms_log ( id bigint generated always as identity primary key, ts bigint not null, sender_name text default '', phone text default '', type text default '', content text default '' ); create table unresolved ( id bigint generated always as identity primary key, ts bigint not null, query text default '', type text default '' ); create table sessions ( id bigint generated always as identity primary key, ts bigint not null, date text default '', messages int default 0, emp text default '', name text default '', dept text default '' ); alter table sms_log enable row level security; alter table unresolved enable row level security; -- If sessions table already exists, run these to add the missing columns: -- alter table sessions add column if not exists emp text default ''; -- alter table sessions add column if not exists name text default ''; -- alter table sessions add column if not exists dept text default ''; alter table sessions enable row level security; create policy "pub read" on sms_log for select using (true); create policy "pub insert" on sms_log for insert with check (true); create policy "pub delete" on sms_log for delete using (true); create policy "pub read" on unresolved for select using (true); create policy "pub insert" on unresolved for insert with check (true); create policy "pub delete" on unresolved for delete using (true); create policy "pub read" on sessions for select using (true); create policy "pub insert" on sessions for insert with check (true); -- Shared key-value store (EMP list, notes, activity log, FAQ edits) create table kv_store ( key text primary key, value text not null, updated_at bigint default 0 ); alter table kv_store enable row level security; create policy "pub read" on kv_store for select using (true); create policy "pub insert" on kv_store for insert with check (true); create policy "pub update" on kv_store for update using (true); create policy "pub delete" on kv_store for delete using (true);
  3. Go to Settings → API, copy Project URL and anon key
  4. Paste below and click Save & Test
  5. Paste the same into index.html in the Supabase Bridge section
